## 1. Vector 자료구조
#######################

# 벡터 데이터 생성 함수
c(1:20) 
seq(1,10,2)
rep(1:3, 3) 

# 벡터 데이터 처리 함수
x <- c(1,3,5) 
y <- c(3,5)
setdiff(x, y) # 차집합
intersect(x, y) # 교집합      

# 벡터 변수에 저장
v1 <- c(33,-5,20:23,12,-2:3)
v2 <-c("홍길동", "이순신", "유관순")
v3 <-c(T, TRUE, FALSE, T, TRUE,F,T)
v1;v2;v3
v4 <- c(33,-5,20:23,12,"4")
v4

# 같은 줄에 명령문 중복사용
v1; mode(v1) 
v2; mode(v2)
v3; mode(v3)
         
# 벡터에 컬럼명 지정
age <- c(10,20,30)
age
names(age) <- c("홍길동", "이순신", "강감찬")
age
age <-NULL # age 변수 데이터 삭제
         
# 특정 요소 출력 및 제외
a <- c(1:50)
a[10 : 45] # 10~45
a[10 : (length(a)-5)] # 10~45
a[1, 2]
